          .from('taxas_clientes')
          .select('*')
          .eq('cliente_id', empresaId)
          .order('vigencia_inicio', { ascending: false })
          .limit(1);
        
        const taxas = taxasArray && taxasArray.length > 0 ? taxasArray[0] : null;
        
        console.log('📊 [buscarTaxaPadraoEmpresa] Resultado da query:', {
          taxas,
          error,
          tem_dados: !!taxas,
          campos_disponiveis: taxas ? Object.keys(taxas) : []
        });
        
        if (error) {
          console.error('❌ [buscarTaxaPadraoEmpresa] Erro ao buscar taxa:', error);
          return;
        }

                         0;

            valor_formatado: new Intl.NumberFormat('pt-BR', {

              currency: 'BRL'

          });
